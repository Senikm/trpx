19// The XML_element class is derived from std::string. It scans an istream for an XML element with a
20// specified tag. The element can contain attibutes and data. The tag itself is stripped and its
21// attributes and data can be accessed as a string or parsed further by constructing another XML_element.
22//
23// After reading the specified Element from istream, the istream is left in the condition that exists
24// after having read the Element, but not further. This allows mixing XML straightforwardly with
25// binary data. For instance, after having read a XML metadata header, the file is positioned where the
26// binary data starts.
27//
28// Constructor:
29// template <typename Stream>
30// XML_element(std::istream& is, std::string const& tag)
31// XML_element(std::istream&& is, std::string const& tag)
32// istr is skipped until the specified XML tag ('<' + tag + '>') is encountered. Then all the XML data
33// defined by that tag, are parsed into the XML_element, so until '</' + tag + '>' is encountered,
34// and scanning the file stops after having read the end tag. Any attributes of the tag are parsed into
35// XML_element. Tags can be empty (when they end with “/>”), in which case only the attributes of tag
36// are parsed into the XML_element.
37// XML_element(std::istream& is)
38// XML_element(std::istream&& is)
39// istr is skipped until the first occurence of a valid XML tag is encountered. Then this tag is used for
40// creating the XML_element.
41// XML_element(std::string const& xml_string)
42// XML_element(std::string const& xml_string, std::string const& tag)
43// Delegates to the constructor requiring a std::istream, after creating a std::istringstream(xml_string).
44//
45// Member functions:
46// std::string const attribute(std::string const& att_name) const
47// Returns the attribute of the tag of XML_element that has the name 'att_name'. An XML_element line does
48// not have to have attributes. If none of the attributes is named 'att_name', an empty string is returned.
49// std::string const attribute(std::size_t n) const
50// Returns the n-th attribute of an XML_element. If n is larger than the number of attributes, an empty
51// string is returned.
52// std::string const tag()
53// Returns the tag name of the XML_element that was used for extracting it from the XML data. If no tag name
54// was specified when the XML_element was created, an empty string is returned.
55// std::string const tag(std::string const& tag)
56// Returns the XML data encoded by the first occurrence of requested tag as an XML_element.
57// std::vector<XML_element> const multitag(std::string const& tag)
58// Returns all requested tags as a vector, in the order in which they occur in original XML data.
59// std::vector<T> const value<T>(std::string const& tag)
60// Returns a vector of the requested type, containing the numerical values of the requested tag of the
61// XML_element.
62// std::vector<std::vector<T>> const multivalue<T>(std::string const& tag)
63// Returns a vector of all the requested value<T>s.
64//
65// An example:
66//
67// int main(int argc, const char * argv[]) {
68// std::string xml("<Element1 att0=\"plop\">\n <Element2 att1=\"1\", att2=\"22\"/>\n</Element1>\n");
69// std::cout << "xml stream:\n" << xml << std::endl <<"Parsing produces:\n";
70// std::cout << "Element 1: " << XML_element(xml, "Element1") << std::endl;
71// std::cout << "Attribute att0 of element 1: " << XML_element(xml, "Element1").attribute("att0") << std::endl;
72// std::cout << "Attribute att1 of element 1 does not exist, so empty string is returned: " << '\"' << XML_element(xml, "Element1").attribute("att1") << '\"' << std::endl;
73// std::cout << "Attribute att1 of element 2: " << XML_element(xml, "Element2").attribute("att1") << std::endl;
74// std::cout << "Attribute att2 of element 2: " << XML_element(xml, "Element2").attribute("att2") << std::endl;
75// std::istringstream xml_stream(xml);
76// XML_element(xml_stream, "Element2");
77// std::string rest;
78// xml_stream >> rest;
79// std::cout << "After reading Element2, the xml_stream is now located ath the end of Element2. \nSubsequent output from theis stream produces:\n"<< rest << std::endl;
80//
81// return 0;
82// }
83//
84// Produces the following output:
85//
86// xml stream:
87// <Element1 att0="plop">
88// <!-- comment " > -->
89// <values> 1 2 3 4 </values>
90// <Element2 att1="1", att2="22"/>
91// </Element1>
92//
93// Parsing produces:
94// Element 1:
95// <!-- comment " > -->
96// <values> 1 2 3 4 </values>
97// <Element2 att1="1", att2="22"/>
98// Attribute att0 of element 1: plop
99// Attribute att1 of element 1 does not exist, so an empty string is returned: ""
100// Attribute att1 of element 2: 1
101// Attribute att2 of element 2: 22
102// After reading Element2, the xml_stream is now located at the end of Element2.
103// Subsequent output from this stream produces:
104// </Element1>

296 std::string const attribute(std::string const& name) const noexcept {
297 for (int i = 0; i <= (d_attributes.size() - name.size() - 3); ++i) {
298 if ((d_attributes[i + name.size()] == '=') && (name == d_attributes.substr(i, name.size()))) {
299 auto strt = 1 + (i += name.size() + 1);

301 return d_attributes.substr(strt, i - strt);
302 }
303 else if (d_attributes[i] == '=')
304 for (char q = d_attributes[++i]; d_attributes[++i] != q;);
305 }
306 return "";

428 std::string f_get_attributes(std::istream& istr) {
429 if (d_tag == "")
430 return "";
431 f_find_tag(istr, d_tag);
432 if (d_tag_found) {
433 std::string r = std::string(f_read_upto(istr, '>'), d_tag.size() + 1);

442 void f_find_tag(std::istream& istr, std::string const& tag) {
443 static auto const all = std::numeric_limits<std::streamsize>::max();
444 for (istr.ignore(all, '<'); istr.good() && !f_next_is(istr, tag); istr.ignore(all, '<')) {
445 if (f_next_is(istr, "![CDATA["))
446 for (f_read_upto(istr, ']'); istr.good() && !f_next_is(istr, "]>"); f_read_upto(istr, ']'));
447 else if (f_next_is(istr, "!--"))
448 for (f_read_upto(istr, '-'); istr.good() && !f_next_is(istr, "->"); f_read_upto(istr, '-'));
449 }
450 if (istr.good())
451 d_tag_found = true;
452 }

454 std::vector<s_param> f_find_comments_and_data(std::string const& s) {
455 std::vector<s_param> r;
456 for (s_param p = f_find_interval(s, "<!--", "-->"); p.from != std::string::npos; p = f_find_interval(s, "<!--", "-->", p.to()))
457 r.push_back(p);
458 for (s_param p = f_find_interval(s, "<![CDATA[", "]]>"); p.from != std::string::npos; p = f_find_interval(s, "<![CDATA[", "]]>", p.to()))
459 r.push_back(p);
460 std::sort(r.begin(), r.end(), [](s_param const& left, s_param const& right) {return left.from < right.from;});
461 for (int i = 0; i!=r.size(); ++i)
462 for (int j = i+1; (j!=r.size()) && (r[j].from < (r[i].from + r[i].size)); ++j)
463 r.erase(r.begin() + j--);
464 return r;
465 }

467 std::vector<s_param> f_find_all_params(std::string const& s) {
468 std::vector<s_param> skip = f_find_comments_and_data(s);
469 std::vector<s_param> r;
470 for (s_param param(*this, f_ignore(s, 0, skip), 1); param.from < s.size(); param.from = f_ignore(s, param.to(), skip)) {
471 if (s[param.from] != '<') {
472 for (param.size = 1; !f_is_white(s[param.to()]) && (s[param.to()] != '<'); ++param.size);
473 r.push_back(param);
474 }
475 else {
476 param.size = s.find('>', param.from) + 1 - param.from;
477 if (s[param.to() - 2] == '/')
478 r.push_back(param);
479 else {
480 std::size_t end_tag_size = s.find_first_of(" >", param.from) - param.from - 1;
481 std::string const end_tag = "</" + s.substr(param.from + 1, end_tag_size) + '>';
482 param.size = s.find(end_tag, param.to()) - param.from + end_tag.size();
483 for (auto& ignore: skip)
484 if ((param.to() > ignore.from) && (param.to() < ignore.to()))
485 param.size = s.find(end_tag, param.to()) - param.from;
486 r.push_back(param);
487 }
488 }
489 }
490 r.insert(r.begin(), skip.begin(), skip.end());
491 std::sort(r.begin(), r.end(), [](s_param const& left, s_param const& right) {return left.from < right.from;});
492 return r;
493 }

495 s_param f_find_interval(std::string const& s, std::string const& from, std::string const& to, std::size_t pos = 0) {
496 std::size_t param_start = s.find(from, pos);
497 if (param_start == std::string::npos)
498 return s_param(*this, param_start, 0);
499 else
500 return s_param(*this, param_start, s.find(to, param_start + from.size()) - param_start + to.size());
501 }

503 std::size_t f_ignore(std::string const& s, std::size_t pos, std::vector<s_param> skip = std::vector<s_param>()) {
504 std::size_t start = pos;
505 std::size_t i = 0;
506 do {
507 pos = start;
508 while (f_is_white(s[start]) && (start < s.size()))
509 ++start;
510 for ( ; (i != skip.size()) && (start >= skip[i].from); ++i)
511 if ((start >= skip[i].from) && (start < skip[i].to()))
512 start = skip[i].to();
513 } while ((pos != start) && (start < s.size()));
514 return start;
515 }

531 std::string f_read_element(std::istream& istr, std::string const& tag) {
532 std::string element = f_read_upto(istr, '<');
533 for (; !f_next_is(istr, '/' + tag + '>'); element += f_read_upto(istr, '<')) {
534 if (f_next_is(istr, "![CDATA["))
535 for (element += f_read_upto(istr, '>'); !element.compare(element.size() - 3, 3, "]]>"); element += f_read_upto(istr, '>'));
536 else if (f_next_is(istr, "!--"))
537 for (element += f_read_upto(istr, '>'); !element.compare(element.size() - 3, 3,"-->"); element += f_read_upto(istr, '>'));
538 }
539 element.pop_back();
540 return element;
541 }

543 bool f_is_white(char const c) const noexcept {return (c == 0x20) || (c==0x09) || (c==0x0D) || (c==0x0A);}

545 bool f_next_is(std::istream& istr, std::string const& s) {
546 std::string tst(s.size(), 0);
547 if (istr.read(&tst[0], s.size())) {
548 istr.seekg(-s.size(), std::ios_base::cur);
549 return d_tag_found = tst == s;
550 }
551 return d_tag_found = false; // Unable to read from the stream, so it's not a match
552 }

554 bool f_prev_is(std::istream& istr, std::string const& s) {
555 std::string tst(s.size(), 0);
556 istr.seekg(-s.size(), std::ios_base::cur);
557 istr.read(&tst[0], s.size());
558 return tst == s;
559 }
