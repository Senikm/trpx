457 template <typename C> requires (requires (C const& c) {std::begin(c), std::end(c), std::size(c);} || std::is_same_v<C, Grey_tif<std::byte>>)
458 void push_back(C const& container, std::array<long,2> dim = {-1,-1}) {
459 if constexpr (is_Grey_tif<C>::value) {
460 for (int i = 0; i < container.image_stack_size(); ++i) {
461 if constexpr (!std::is_same_v<std::byte, std::remove_const_t<typename C::value_type>>)
462 push_back(container.d_img[i]);
463 else {
464 POD_type_traits const& img_type = container.image(i).type();
465 if (img_type.is<std::int8_t>()) push_back(Grey_tif_image<std::int8_t>(container.d_img[i]));
466 else if (img_type.is<std::uint8_t>()) push_back(Grey_tif_image<std::uint8_t>(container.d_img[i]));
467 else if (img_type.is<std::int16_t>()) push_back(Grey_tif_image<std::int16_t>(container.d_img[i]));
468 else if (img_type.is<std::uint16_t>()) push_back(Grey_tif_image<std::uint16_t>(container.d_img[i]));
469 else if (img_type.is<std::int32_t>()) push_back(Grey_tif_image<std::int32_t>(container.d_img[i]));
470 else if (img_type.is<std::uint32_t>()) push_back(Grey_tif_image<std::uint32_t>(container.d_img[i]));
471 else if (img_type.is<float>()) push_back(Grey_tif_image<float>(container.d_img[i]));
472 else if (img_type.is<double>()) push_back(Grey_tif_image<double>(container.d_img[i]));
473 }
474 }
475 }
476 else {
477 if constexpr (requires(C const& c) { c.dim().size();})
478 if (dim[0] == -1 && dim[1] == -1) {
479 assert(container.dim().size() == 2); // Grey_tif.pushback(container) only accepts 2D containers
480 dim[0] = container.dim()[0];
481 dim[1] = container.dim()[1];
482 }
483 using CT = typename C::value_type;
484 static_assert ((std::is_arithmetic_v<CT> && sizeof(CT) <= 8 && !std::is_same_v<CT, bool> && !(sizeof(CT) == 8 && std::is_integral_v<CT>)) ||
485 std::is_same_v<float, CT> || std::is_same_v<double, CT>);
486 assert(container.size() == dim[0] * dim[1]);
487 assert((d_tif.size() + dim[0] * dim[1] + 6 * 12 + 4) < std::numeric_limits<uint32_t>::max());
488 std::uint32_t index = static_cast<uint32_t>(d_tif.size());
489 std::uint32_t data_start = index;
490 if constexpr (std::is_same_v<T, std::byte>) {
491 d_tif.resize(d_tif.size() + dim[0] * dim[1] * sizeof(CT) + 7 * 12 + 6);
492 for (auto const val : container) {
493 reinterpret_cast<CT&>(d_tif[index]) = static_cast<CT>(val);
494 index += sizeof(CT);
495 }
496 }
497 else {
498 d_tif.resize(d_tif.size() + dim[0] * dim[1] * sizeof(T) + 7 * 12 + 6);
499 for (auto const val : container) {
500 reinterpret_cast<T&>(d_tif[index]) = static_cast<T>(val);
501 index += sizeof(T);
502 }
503 }
504 if ((d_tif.size() & 1) == 1) {
505 d_tif.push_back(std::byte(0));
506 ++index;
507 }
508 reinterpret_cast<std::uint32_t&>(d_tif[d_last_ifd_offset]) = index;
509 reinterpret_cast<std::uint16_t&>(d_tif[index]) = 7;
510 index += 2;
511 f_set_ifd(index, 0x0100, 3, static_cast<uint32_t>(dim[0]));
512 f_set_ifd(index, 0x0101, 3, static_cast<uint32_t>(dim[1]));
513 f_set_ifd(index, 0x0102, 3, 8 * (std::is_same_v<std::byte, T> ? sizeof(CT) : sizeof(T)));
514 f_set_ifd(index, 0x0103, 3, 1);
515 f_set_ifd(index, 0x0106, 3, 1);
516 f_set_ifd(index, 0x0111, 4, data_start);
517 if constexpr (std::is_same_v<std::byte, T>) {
518 if (std::is_unsigned_v<CT> )
519 f_set_ifd(index, 0x0153, 3, 1);
520 else if (std::is_integral_v<CT>)
521 f_set_ifd(index, 0x0153, 3, 2);
522 else if (std::is_floating_point_v<CT>)
523 f_set_ifd(index, 0x0153, 3, 3);
524 }
525 else {
526 if (std::is_unsigned_v<T> )
527 f_set_ifd(index, 0x0153, 3, 1);
528 else if (std::is_integral_v<T>)
529 f_set_ifd(index, 0x0153, 3, 2);
530 else if (std::is_floating_point_v<T>)
531 f_set_ifd(index, 0x0153, 3, 3);
532 }
533 d_tif[index] = d_tif[index + 1] = d_tif[index + 2] = d_tif[index + 3] = std::byte(0);
534 d_last_ifd_offset = static_cast<uint32_t>(index);
535 f_scan_images();
536 }

580 // Push_back an all-zero image of the specified type
581 void f_push_back(std::array<long,2> const& dim, POD_type_traits const& new_type) {
582 assert((d_tif.size() + dim[0] * dim[1] + 6 * 12 + 4) < std::numeric_limits<uint32_t>::max());
583 assert((this->type().template is<T>() || std::is_same_v<T, std::byte>));
584 std::uint32_t index = static_cast<uint32_t>(d_tif.size());
585 std::uint32_t data_start = index;
586 d_tif.resize(d_tif.size() + dim[0] * dim[1] * new_type.size + 7 * 12 + 6);
587 index += dim[0] * dim[1] * new_type.size;
588 if ((d_tif.size() & 1) == 1) {
589 d_tif.push_back(std::byte(0));
590 ++index;
591 }
592 reinterpret_cast<std::uint32_t&>(d_tif[d_last_ifd_offset]) = index;
593 reinterpret_cast<std::uint16_t&>(d_tif[index]) = 7;
594 index += 2;
595 f_set_ifd(index, 0x0100, 3, static_cast<uint32_t>(dim[0]));
596 f_set_ifd(index, 0x0101, 3, static_cast<uint32_t>(dim[1]));
597 f_set_ifd(index, 0x0102, 3, 8 * static_cast<uint32_t>(new_type.size));
598 f_set_ifd(index, 0x0103, 3, 1);
599 f_set_ifd(index, 0x0106, 3, 1);
600 f_set_ifd(index, 0x0111, 4, data_start);
601 f_set_ifd(index, 0x0153, 3, new_type.is_integral ? (!new_type.is_signed ? 1 : 2) : 3);
602 d_tif[index] = d_tif[index + 1] = d_tif[index + 2] = d_tif[index + 3] = std::byte(0);
603 d_last_ifd_offset = static_cast<uint32_t>(index);
604 f_scan_images();
605 }

607 Grey_tif& f_regularize() noexcept {
608 static_assert(!std::is_same_v<T, std::byte>);
609 POD_type_traits new_type = {.size = sizeof(T), .is_signed = std::is_signed_v<T>, .is_integral = std::is_integral_v<T> };
610 if (f_all_same_type(new_type))
611 return *this;
612 else {
613 bool same_size = this->type().size == new_type.size;
614 for (auto it = d_img.begin() + 1; it != d_img.end() && same_size == true; ++it)
615 same_size = it->type().size == new_type.size;
616 if (same_size) {
617 for (auto& img : d_img) {
618 if (!new_type.is_integral && img.type().is_integral && img.type().is_signed)
619 for (auto& pixel : img)
620 reinterpret_cast<float&>(pixel) = static_cast<float>(reinterpret_cast<int&>(pixel));
621 else if (!new_type.is_integral && img.type().is_integral && !img.type().is_signed)
622 for (auto& pixel : img)
623 reinterpret_cast<float&>(pixel) = static_cast<float>(reinterpret_cast<unsigned int&>(pixel));
624 else if (new_type.is_integral && new_type.is_signed && !img.type().is_integral)
625 for (auto& pixel : img)
626 reinterpret_cast<std::int32_t&>(pixel) = static_cast<int>(reinterpret_cast<float&>(pixel));
627 else if (new_type.is_integral && !new_type.is_signed && !img.type().is_integral)
628 for (auto& pixel : img)
629 reinterpret_cast<std::uint32_t&>(pixel) = static_cast<int>(reinterpret_cast<float&>(pixel));
630 img = Grey_tif_image(new_type, img.dim(), img);
631 }
632 }
633 else {
634 Grey_tif new_tif;
635 for (int i=0; i != image_stack_size(); ++i) {
636 new_tif.f_push_back(image(i).dim(), new_type);
637 POD_type_traits const& old_type = image(i).type();
638 std::size_t size = image(i).dim()[0] * image(i).dim()[1];
639 auto new_begin = new_tif.image(i).begin();
640 if (old_type.is<std::int8_t>()) std::copy_n(reinterpret_cast<std::int8_t*>(&d_img[i][0]) , size, new_begin);
641 else if (old_type.is<std::uint8_t>()) std::copy_n(reinterpret_cast<std::uint8_t*>(&d_img[i][0]) , size, new_begin);
642 else if (old_type.is<std::int16_t>()) std::copy_n(reinterpret_cast<std::int16_t*>(&d_img[i][0]) , size, new_begin);
643 else if (old_type.is<std::uint16_t>()) std::copy_n(reinterpret_cast<std::uint16_t*>(&d_img[i][0]) , size, new_begin);
644 else if (old_type.is<std::int32_t>()) std::copy_n(reinterpret_cast<std::int32_t*>(&d_img[i][0]) , size, new_begin);
645 else if (old_type.is<std::uint32_t>()) std::copy_n(reinterpret_cast<std::uint32_t*>(&d_img[i][0]) , size, new_begin);
646 else if (old_type.is<float>()) std::copy_n(reinterpret_cast<float*>(&d_img[i][0]) , size, new_begin);
647 else if (old_type.is<double>()) std::copy_n(reinterpret_cast<double*>(&d_img[i][0]) , size, new_begin);
648 }
649 std::swap(new_tif, *this);
650 }
651 return *this;
652 }

655 void f_scan_images() {
656 d_img.clear();
657 d_img_const.clear();
658 this->d_dim = {0,0};
659 bool native = d_tif[0] == std::byte('I') && std::endian::native == std::endian::little;
660 if (!native)
661 d_tif[0] = d_tif[1] = (d_tif[0] == std::byte('M')) ? std::byte('I'): std::byte('M');
662 reinterpret_cast<std::uint16_t&>(d_tif[2]) = 42;
663 std::uint32_t index = reinterpret_cast<std::uint32_t&>(d_tif[4]);
664 if (!native) {
665 index = (index << 24) | ((index & 0xff00) << 8) | ((index >> 8) & 0xff00) | (index >> 24);
666 reinterpret_cast<std::uint32_t&>(d_tif[4]) = index;
667 }
668 if (native)
669 while (index != 0) {
670 f_make_Image<true>(index);
671 index = reinterpret_cast<std::uint32_t&>(d_tif[d_last_ifd_offset]);
672 }
673 else
674 while (index != 0) {
675 f_make_Image<false>(index);
676 index = reinterpret_cast<std::uint32_t&>(d_tif[d_last_ifd_offset]);
677 }
678 if (image_stack_size() != 0) {
679 if constexpr(!std::is_same_v<T, std::byte>)
680 std::span<T const>::operator=(d_img_const[0]);
681 else
682 this->d_data = d_img_const[0].d_data;
683 this->d_dim = d_img[0].dim();
684 this->d_image_type = d_img[0].type();
685 }
686 if constexpr (!std::is_same_v<T, std::byte>)
687 f_regularize();
688 }

690 template <bool NATIVE = true>
691 void f_make_Image(std::uint32_t& index) noexcept {
692 std::array<long,2> dim = {0,0};
693 std::size_t bits_per_pixel = 0;
694 std::byte* cursor = d_tif.data() + index;
695 int const tag_count = f_int16<NATIVE>(cursor);
696 std::int32_t rows_per_strip;
697 std::vector<std::int32_t> strip_offsets(1);
698 std::vector<std::int32_t> strip_byte_counts(1);
699 bool signed_pixels = false;
700 bool int_pixels = true;
701 for (int i = 0; i != tag_count; ++i) {
702 uint16_t tag = f_int16<NATIVE>(cursor);
703 uint16_t type = f_int16<NATIVE>(cursor);
704 uint32_t count = f_int32<NATIVE>(cursor);
705 uint32_t val(0);
706 double double_val(0);
707 if ((type == 1) || (type == 2) || (type == 6) || (type == 7)) {
708 val = f_int8(cursor);
709 cursor += 3;
710 }
711 else if ((type == 3) || (type == 8)) {
712 val = f_int16<NATIVE>(cursor);
713 cursor += 2;
714 }
715 else if ((type == 4) || (type == 9))
716 val = f_int32<NATIVE>(cursor);
717 else if (type == 5 || type == 10) {
718 std::byte* p = d_tif.data() + f_int32<NATIVE>(cursor);
719 double_val = f_int32<NATIVE>(p);
720 double_val /= f_int32<NATIVE>(p);
721 }
722 else if (type == 11) {
723 val = f_int32<NATIVE>(cursor);
724 double_val = reinterpret_cast<float&>(val);
725 val = 0;
726 }
727 else if (type == 12) {
728 std::byte* p = d_tif.data() + f_int32<NATIVE>(cursor);
729 std::uint64_t tmp = f_int64<NATIVE>(p);
730 double_val = reinterpret_cast<double&>(tmp);
731 }
732 if (tag == 0x0100) dim[0] = val;
733 else if (tag == 0x0101) dim[1] = val;
734 else if (tag == 0x0102 ) {
735 if ((val == 8 || val == 16 || val == 32 || val == 64))
736 bits_per_pixel = val;
737 else
738 std::cerr << "Warning: Grey_tif can only read greyscale tiff files with 8-, 16-, 32-, or 64-bit pixels" << std::endl;
739 }
740 else if (tag == 0x0103 && val != 1)
741 std::cerr << "Warning: Grey_tif cannot read compressed tiff files"<< std::endl;
742 else if (tag == 0x0106 && val > 1)
743 std::cerr << "Warning: Grey_tif cannot read colour tiff files"<< std::endl;
744 else if ((tag == 0x0107 || tag == 0x0108 || tag == 0x0109 || tag == 0x010A) && val != 1)
745 std::cerr << "Warning: Grey_tif cannot read black & white tiff files"<< std::endl;
746 else if (tag == 0x0111) {
747 if (count == 1) strip_offsets[0] = val;
748 else {
749 strip_offsets.resize(count);
750 std::byte* p = d_tif.data() + val;
751 for (int i=0; i != count; ++i)
752 strip_offsets[i] = f_int32<NATIVE>(p);
753 }
754 }
755 else if (tag == 0x0115 && val != 1)
756 std::cerr << "Warning: Grey_tif cannot read RGB colour tiff files"<< std::endl;
757 else if (tag == 0x0116) rows_per_strip = val;
758 else if (tag == 0x0117) {
759 if (count == 1) strip_byte_counts[0] = val;
760 else {
761 strip_byte_counts.resize(count);
762 std::byte* p = d_tif.data() + val;
763 for (int i=0; i != count; ++i)
764 strip_byte_counts[i] = f_int32<NATIVE>(p);
765 }
766 }
767 else if (tag == 0x0153) {
768 if (val != 1) signed_pixels = true;
769 if (val == 3) int_pixels = false;
770 }
771 }
772 for (int i=0; i != strip_offsets.size() - 1; ++i)
773 if (strip_byte_counts[i] != strip_offsets[i + 1] - strip_offsets[i]) {
774 std::cerr << "Warning: Grey_tif cannot read tiff files with non-consecutive strips" << std::endl;
775 std::cerr << " most likely the tiff file is corrupted" << std::endl;
776 break;
777 }
778 d_last_ifd_offset = static_cast<std::uint32_t>(cursor - d_tif.data());
779 index = f_int32<NATIVE>(cursor);
780 if constexpr (!NATIVE) {
781 std::byte* begin = d_tif.data() + strip_offsets[0];
782 std::byte* end = begin + dim[0] * dim[1] * bits_per_pixel / 8;
783 if (bits_per_pixel == 16)
784 while (begin != end) f_int16<false>(begin);
785 else if (bits_per_pixel == 32)
786 while (begin != end) f_int32<false>(begin);
787 else if (bits_per_pixel == 64)
788 while (begin != end) f_int64<false>(begin);
789 }
790 POD_type_traits image_type = { .size = bits_per_pixel / 8, .is_signed = signed_pixels, .is_integral = int_pixels };
791 d_img.push_back(Grey_tif_image<T>(image_type, dim, std::span<T>(reinterpret_cast<T*>(&d_tif[strip_offsets[0]]), dim[0] * dim[1])));
792 d_img_const.push_back(Grey_tif_image<T const>(image_type, dim, std::span<T const>(reinterpret_cast<T const*>(&d_tif[strip_offsets[0]]), dim[0] * dim[1])));
793 };

829 cursor += 8;
830 return NATIVE ? r : ((r << 56) | ((r & 0xff00) << 40) | ((r & 0xff0000) << 24) | ((r & 0xff000000) << 8) |
831 ((r >> 8) & 0xff000000) | ((r >> 24) & 0xff0000) | ((r >> 40) & 0xff00) | (r >> 56));
