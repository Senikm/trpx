19// Terse<T> allows efficient and fast compression of integral diffraction data and other integral greyscale
20// data into a Terse object that can be decoded by the member function Terse<T>::prolix(iterator). The
21// prolix(iterator) member function decompresses the data starting at the location defined by 'iterator'
22// (which can also be a pointer). A Terse object is constructed by supplying it with uncompressed data or a
23// stream that contains compressed Terse data.
24//
25// A Terse object may contain compressed data of multiple frames, and data of a particular frame can be
26// extracted by indexing. All frames must have the same size and dimensions, and must all be signed or unsigned.
27//
28// A Terse object can be unpacked into any arithmetic type, including float and double. Unpacking into
29// values of type T with fewer bits than the original data, results in truncation of overflowed data to
30// std::numeric_limits<T>::max(),and to std::numeric_limits<T>::min() for underflowed signed types.
31// Unpacking signed data into unsigned values is not allowed. Compressing as unsigned yields a tighter
32// compression.
33//
34// A Terse object can be written or appended to any stream. The resulting file is independent of the endian-nes
35// of the machine: both big- and small-endian machines produce identical files, making data transfer optimally
36// transparent.
37//
38// Terse data in a file are immediately preceded by a small header, which is encoded in standard XML as follows:
39// <Terse prolix_bits="n" signed="s" block="b" memory_size="m" number_of_values="v" [dimensions="d [...]"] [number_of_frams="f"]/>
40// - "n" is the number of bits required for the most extreme value in the Terse data
41// - "s" is "0" for unsigned data, "1" for signed data
42// - "b" is the block size of the stretches of data values that are encoded (by default 12 values)
43// - "m" is the number of bytes of Terse data, excluding the header, but including all frames in an encoded stack
44// - "v" is the number of values of a single frame of a stack
45// - "d [...]" is optional. It encodes the dimensions of a single frame. Frames can have any number of dimensions.
46// - "f" is optional. if it is absent, a single frame is encoded, if it is present, f indicates the number of frames.
47// Here is an example:
48// <Terse prolix_bits="12" signed="0" block="12" memory_size="91388" number_of_values="262144" dimensions="512 512" number_of_frames="2"/>
49//
50// The algorithm is a run-length encoding type. Each data block (by default 12 integral values in the
51// constructor, but this can be changed) is preceded by one or more data block header bits. The values in the
52// data block are stripped of their most significant bits, provided they are all zero (for unsigned values),
53// or either all zero or one (for signed values). In the latter case, the sign bit is maintained. So, for a
54// block size of 3 with values 3, 4, 2, the encoded bits would be: 011 (denoting 3) 100 (denoting 4) 010
55// (denoting 2). So 011100010 would be pushed into the Terse object. In case of signed values -3, 4, 2, the
56// encoded bits would be 1011 (denoting -3) 0100 (denoting +4) 0010 (denoting +2), resulting in a data block
57// 101101000010. So if the values that need to be encoded are all positive or zero, they should be encoded as
58// unsigned for optimal compression: it saves 1 bit per encoded value.
59//
60// The header bits define how the values are encoded. They have the following following structure:
61// bit 1: If the first bit of the block header is set, then there are no more bits in the block header,
62// and the parameters of the previous block header are used.
63// bit 2-4: The first header bit is 0. The three bits 2 to 4 define how many bits are used per value of the
64// encoded block. If bits 2 to 4 are all set, 7 or more bits per value are required and the
65// header is expanded by a further 2 bits.
66// bit 5-6: The first 4 header bits are 0111. The number encoded bits 5 and 6 is added to 7 and this defines
67// how many bits are used to encode the block. So if bits 5 & 6 are 00 then 7 bits are used, 01 means
68// 8 bits, 10 means 9 bits and 11 means at least 10 bits. If bits 5 & 6 are both set, the header is
69// expanded by another 6 bits.
70// bit 7-12: The first 6 header bits are 011111. The number encoded by bits 7 to 12 is added to decimal 10 and
71// this defines how many bits are used to encode the block. So if bits 7 to 12 are 000000, then the
72// number of bits per value in the data block is decimal 10. If bits 7 to 12 are 110110, then the
73// number of bits per value in the data block is 10 + 54 = 64.
74//
75// Constructors:
76// Terse(std::ifstream& istream)
77// Reads in a Terse object that has been written to a file by the overloaded Terse output operator '<<'.
78// Terse(container_type const& data)
79// Creates a Terse object from data (which can be a std::vector, Field, etc.). Only containers of
80// integral types are allowed. If the container has a member function dim(), that will set the dimensions
81// of the Terse object. Otherwise the dimensions can be set once using the dim(vector const&) member function.
82// Terse(iterator begin, std::size_t size)
83// Creates a Terse object given a starting iterator or pointer and the number of elements that need to be
84// encoded.
85//
86// Member functions:
87// std::size_t size()
88// Returns the number of encoded elements.
89// bool is_signed()
90// Returns true if the encoded data are signed, false if unsigned. Signed data cannot be decompressed into
91// unsigned data.
92// bits_per_val()
93// Returns the maximum number of bits per element that can be expected. So for uncompressed uint_16 type
94// data, bits_per_val() returns 16. Terse data cannot be decompressed into a container type with elements
95// that are smaller in bits than bits_per_val().
96// terse_size()
97// Returns the number of bytes used for encoding the Terse data.
98// void push_back(Iterator const begin, size_t const size)
99// void push_back(container_type const& container)
100// Adds another frame to the Terse object. The new frame is defined by its begin iterator and size, or by a
101// reference to a container. The size must be the same as that of the frame used to create the Terse object.
102// If the container has a member function dim(), that must return the same dimension as provided for the first frame.
103// std::size_t const number_of_frames() const
104// Returns the number of frames stored in the Terse object
105// std::vector<std::size_t> const& dim() const
106// Returns the dimensions of each of the Terse frames (all frames must have the same dimensions).
107// std::vector<std::size_t> const& dim(std::vector<std::size_t> const& dim) {
108// Sets the dimensions of the Terse frames. Since all frames must have the same dimensions, they can be set only once.
109// void prolix(iterator begin)
110// Unpacks the Terse data, storing it from the location defined by 'begin'. Terse integral signed data cannot be
111// unpacked into integral unsigned data. Terse data cannot be decompressed into elements that are smaller
112// in bits than bits_per_val(), but can be decompressed into larger values. Terse data can always be unpacked
113// into signed intergral, double and float data and will have the correct sign (with one exception: an
114// unsigned overflowed - all 1's - value will be unpacked as -1 signed value. As all other values are positive
115// in this case, such a situation is easy to recognise).
116// void prolix(container_type& container)
117// Unpacks the Terse data and stores it in the provided container. Also checks the container is large enough.
118// void write(Streamtype &ostream)
119// Writes Terse data to 'ostream'. The Terse data are preceded by an XML element containing the parameters
120// that are required for constructing a Terse object from the stream. Data are written as a byte stream
121// and are therefore independent of endian-ness. A small-endian memory lay-out produces the a Terse file
122// that is identical to a big-endian machine.
123//
124//
125// Example:
126//
127// std::vector<int> numbers(1000); // Uncompressed data location
128// std::iota(numbers.begin(), numbers.end(), -500); // Fill with numbers -500, -499, ..., 499
129// Terse trpx(numbers); // Compress the data to less than 30% of memory
130// std::cout << "compression rate " << float(compressed.terse_size()) / (numbers.size() * sizeof(unsigned)) << std::endl;
131// std::ofstream outfile("junk.trpx");
132// compressed.write(outfile); // Write Terse data to disk
133// std::ifstream infile("junk.trpx");
134// Terse from_file(infile); // Read it back in again
135// std::vector<int> uncompressed(1000);
136// from_file.prolix(uncompressed.begin()); // Decompress the data...
137// for (int i=0; i != 5; ++i)
138// std::cout << uncompressed[i] << std::endl;
139// for (int i=995; i != 1000; ++i)
140// std::cout << uncompressed[i] << std::endl;
141//
142// Produces as output:
143//
144// compression rate 0.29
145// -500
146// -499
147// -498
148// -497
149// -496
150// 495
151// 496
152// 497
153// 498

352 template <typename Iterator> requires requires (Iterator& i) {*i;}
353 void prolix(Iterator begin, std::size_t frame = 0) {
354 assert(frame < number_of_frames());
355 std::uint8_t const* terse_begin = f_find_terse_frame(frame);
356 if (d_signed)
357 assert(std::is_signed_v<typename std::iterator_traits<Iterator>::value_type>);
358 Bit_pointer bitp(terse_begin);
359 uint8_t significant_bits = 0;
360 for (size_t from = 0; from < size(); from += d_block) {
361 if (*bitp++ == 0) {
362 significant_bits = Bit_range<const std::uint8_t*>(bitp,3);
363 bitp += 3;
364 if (significant_bits == 7) {
365 significant_bits += uint8_t(Bit_range<const std::uint8_t*>(bitp, 2));
366 bitp += 2;
367 if (significant_bits == 10) {
368 significant_bits += uint8_t(Bit_range<const std::uint8_t*>(bitp, 6));
369 bitp += 6;
370 }
371 }
372 }
373 if (significant_bits == 0)
374 std::fill(begin + from, begin + std::min(size(), from + d_block), 0);
375 else {
376 Bit_range<const std::uint8_t*> bitr(bitp, significant_bits);
377 if constexpr (std::is_integral<typename std::iterator_traits<Iterator>::value_type>::value)
378 bitr.get_range(begin + from, begin + std::min(size(), from + d_block));
379 else if (!is_signed())
380 for (auto i = from, to = std::min(size(), from + d_block); i < to; ++i, bitr.next())
381 begin[i] = double(std::uint64_t(bitr));
382 else for (auto i = from, to = std::min(size(), from + d_block); i < to; ++i, bitr.next())
383 begin[i] = double(std::int64_t(bitr));
384 bitp = bitr.begin();
385 }
386 }
387 if (d_terse_frames.size() > ++frame)
388 d_terse_frames[frame] = 1 + (bitp - Bit_pointer<const std::uint8_t*>(terse_begin)) / 8;
389 }

500 template <typename Iterator>
501 void const f_compress(Iterator data) {
502 std::size_t prev_data_size = d_terse_data.size();
503 d_terse_data.resize(prev_data_size + std::ceil(d_size * (sizeof(decltype(*data)) + (long double)(12.0) / (d_block * 8)) / sizeof(std::uint8_t)), 0);
504 Bit_pointer bitp (d_terse_data.data() + prev_data_size);
505 int prevbits = 0;
506 for (size_t from = 0; from < d_size; from += d_block) {
507 auto const to = std::min(d_size, from + d_block);
508 typename std::iterator_traits<Iterator>::value_type setbits(0);
509 auto p = data;
510 for (auto i = from; i != to; ++i, ++p)
511 if constexpr (std::is_unsigned_v<decltype(setbits)>)
512 setbits |= *p;
513 else if constexpr (std::is_signed_v<decltype(setbits)>)
514 setbits |= std::abs(*p);
515 unsigned significant_bits = f_highest_set_bit(setbits);
516 d_prolix_bits = std::max(d_prolix_bits, significant_bits);
517 if (prevbits == significant_bits) {
518 (*bitp).set();
519 ++bitp;
520 }
521 else {
522 if (significant_bits < 7) {
523 Bit_range<std::uint8_t*>(++bitp, 3) |= significant_bits;
524 bitp += 3;
525 }
526 else if (significant_bits < 10) {
527 Bit_range<std::uint8_t*>(++bitp, 5) |= 0b111 + ((significant_bits - 7) << 3);
528 bitp += 5;
529 }
530 else {
531 Bit_range<std::uint8_t*>(++bitp, 11) |= 0b11111 + ((significant_bits - 10) << 5);
532 bitp += 11;
533 }
534 prevbits = significant_bits;
535 }
536 if (significant_bits != 0) {
537 Bit_range<std::uint8_t*> r(bitp, significant_bits);
538 r.append_range(data, data + (to - from));
539 data += (to - from);
540 bitp = r.begin();
541 }
542 else if constexpr (std::is_same_v<std::random_access_iterator_tag, typename std::iterator_traits<Iterator>::iterator_category>)
543 data += d_block;
544 else
545 for (int i = 0; i != d_block; ++i, ++data);
546 }
547 d_terse_data.resize(1 + (bitp - d_terse_data.data()) / (sizeof(std::uint8_t) * 8));
548 d_terse_data.shrink_to_fit();
549 }

551 template <typename T0>
552 constexpr inline int const f_highest_set_bit(T0 val) const noexcept {
553 if constexpr (std::is_signed_v<T0>)
554 return (val == 0) ? 0 : 1 + f_highest_set_bit(std::make_unsigned_t<T0> (abs(val)));
555 else {
556 int r=0;
557 for ( ; val; val>>=1, ++r);
558 return r;
559 }
560 }

562 std::uint8_t const* f_find_terse_frame(std::size_t frame) {
563 if (frame > 0 && d_terse_frames[frame] == 0) {
564 std::uint8_t const* terse_begin = f_find_terse_frame(frame - 1);
565 Bit_pointer<const std::uint8_t*> bitp(terse_begin);
566 uint8_t significant_bits = 0;
567 for (size_t from = 0; from < size(); from += d_block) {
568 if (*bitp++ == 0) {
569 significant_bits = Bit_range<const std::uint8_t*>(bitp,3);
570 bitp += 3;
571 if (significant_bits == 7) {
572 significant_bits += uint8_t(Bit_range<const std::uint8_t*>(bitp, 2));
573 bitp += 2;
574 if (significant_bits == 10) {
575 significant_bits += uint8_t(Bit_range<const std::uint8_t*>(bitp, 6));
576 bitp += 6;
577 }
578 }
579 }
580 bitp += significant_bits * d_block;
581 }
582 d_terse_frames[frame] = 1 + (bitp - Bit_pointer<const std::uint8_t*>(terse_begin)) / 8;
583 }
584 return d_terse_data.data() + d_terse_frames[frame];
585 }
