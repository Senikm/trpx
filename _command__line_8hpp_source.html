18// Command_line.hpp is a light-weight parser specifically created for parsing (argc, *argv[]) command line parameters.
19//
20// It defines two classes: Command_line and Command_line_option.
21//
22// Command_line_option:
23// Command_line_option(std::string const& option, std::string const& help, std::vector<std::string> const& default_params = std::vector<std::string>())
24// Constructor that takes a option, a help string describing the option and a vector<string> with default arguments as parameters.
25// The default_params define the default values of the parameters associated with the specified option, and also the number of
26// of allowed parameters. The default string ".*" indicates that the number of parameters is not fixed and could be zero.
27// It is derived from std::string, which contains the input option.
28// template <typename T> std::vector<T> param(std::size_t indx = 0) const

30// not included in the command line, it will get the default value, unless the parameter is mandatory (so with
31// an empty default string), in which case an exception is thrown.
32// template <typename T> std::vector<T> all_params(std::size_t indx = 0) const
33// Type-cast the Command_line_option parameter with the supplied index to type T and return it. If a parameter is
34// not included in the command line, it will get the default value, unless the parameter is mandatory (so with
35// an empty default string), in which case an exception is thrown.
36// bool const found() const
37// Returns true if the Command_line_option was found in a Command_line.
38// std::vector<std::ptrdiff_t> const positions() const

40//
41// Example:
42// Command_line_option option0 {"-opt0", "two integers", {"0", "10"}};
43// Defines a Command_line_option with "-opt0" as option, a help string and two parameters, with default "0" and "10"
44// Command_line_option option1{"-opt1", "two mandatory integers", {"", ""}};
45// Defines a Command_line_option with "-opt1" as the option, a help string, and two parameters with default values "0" and "10".

181 template <typename T> std::vector<T> param(std::size_t indx = 0) const {
182 std::vector<T> r;
183 if (found())
184 try {
185 assert (indx < d_values.size());
186 for (auto &val : d_values[indx])
187 r.push_back(val);
188 }
189 catch(std::string const& msg) {
190 std::cerr << msg << " in:\n " + *this + " ";
191 for(auto &val : d_values[indx])
192 std::cerr << val + " ";
193 std::cerr << std::endl;
194 }
195 else {
196 assert (indx == 0);
197 for(auto &val : d_default_params)
198 r.push_back(val);
199 }
200 return r;
201 }

300class Command_line : private std::vector<std::string> {
301public:
310 Command_line(int argc, const char *argv[], std::vector<Command_line_option> const& options) :
311 vector([&]{
312 vector<std::string> r;
313 for (int i=0; i<argc; ++i)
314 r.push_back(argv[i]);
315 return r;
316 }()),
317 d_options(options)
318 {
319 for (auto& option : d_options) {
320 for (auto cursor = std::find(begin(), end(), option); cursor != end(); cursor = std::find(cursor, end(), option)) {
321 option.d_found_at.push_back(cursor - begin());
322 option.d_values.push_back(std::vector<Command_line_option::Key_value>());
323 auto next_option_p = std::find_first_of(++cursor, end(), d_options.begin(), d_options.end());
324 if (option.d_default_params.size() != 0 && option.d_default_params[0] == ".*")
325 while (cursor != next_option_p)
326 option.d_values.back().push_back(*(cursor++));
327 else {
328 for (auto& default_val : option.d_default_params)
329 option.d_values.back().push_back(default_val);
330 std::copy(cursor, std::min(cursor + option.d_default_params.size(), next_option_p), option.d_values.back().begin());
331 }
332 auto missing_values = option.d_default_params.size();
333 for (auto const& v : option.d_values.back())
334 missing_values -= v.size() == 0 ? 0 : 1;
335 if (missing_values == 1)
336 throw "There is 1 mandatory value missing for command line option " + option;
337 else if (missing_values != 0)
338 throw "There are " + std::to_string(missing_values) + " mandatory values missing for command line option " + option;
339 }
340 }
341 for (auto param = this->begin() + 1; param != this->end(); ) {
342 Command_line_option opt = this->option(*param);
343 if (opt == "")
344 d_parameters.push_back(*(param++));
345 else {
346 auto next_opt = std::find_first_of(++param, end(), d_options.begin(), d_options.end());
347 if (opt.d_default_params.size() != 0 && opt.d_default_params[0] == ".*")
348 param = next_opt;
349 else
350 param = std::min(param + opt.d_default_params.size(), next_opt);
351 }
352 }
353 }
354
363 Command_line_option const& option(std::string const& option_name) const noexcept {
364 auto r = std::find(d_options.begin(), d_options.end(), option_name);
365 if (r != d_options.end())
366 return *std::find(d_options.begin(), d_options.end(), option_name);
367 return d_invalid;
368 }

375 std::vector<std::string> const& arguments() const noexcept { return *this; }

382 std::string const help() const noexcept {
383 std::string r;
384 for (auto& k: d_options) {
385 r += static_cast<std::string>(k) + ": " + k.d_help + '\n';
386 if (k.d_default_params.size() > 0) {
387 r += " default: ";
388 for (auto& p: k.d_default_params) {
389 if (p == "")
390 r += "(no default: mandatory parameter) ";
391 else
392 r += p + ' ';
393 }
394 if (k.d_default_params[0] == ".*")
395 r += " (number of parameters unspecified; no parameters by default)";
396 r += '\n';
397 }
398 }
399 return r;
400 }
