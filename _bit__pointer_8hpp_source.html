17// Bit<T>, Bit_pointer<T> and Bit_range<T> (where T is a random access iterator or pointer referring
18// to an integral type), are three classes that provide a more versatile and powerful alternative to
19// std::bitset and std::vector<bool>. Unlike, the number of bits is not limited to 64. Unlike
20// std::vector<bool>, these classes allow direct random access any type of container, where the bits
21// are guaranteed to be consecutive.
22//
23// Bit<T> is a reference to a bit in an object of type *T. It allows retrieving and manipulating the
24// referenced bit. It has the following member functions:
25//
26// operator bool()
27// Returns the value of the referenced bit by implicit typecasting to bool.
28// bool const operator =(bool const val)
29// Sets the referenced bit to val.
30// bool const operator =(Bit& const other)
31// Sets the referenced bit to the value of other.
32// Bit_pointer<Iter> operator &()
33// Returns a Bit_pointer<Iter> to the referenced bit.
34// void set()
35// Sets the referenced bit to 1 / 'true'.
36// void reset()
37// Sets the referenced bit to 0 / 'false'.
38// void flip()
39// Flips the referenced bit to from 1 to 0 or vice versa.
40// void swap(Bit_reference& other)
41// Swaps the bit referenced to by *this with the bit referenced to by other.
42// Bit<Iter>& operator &=(bool b) const noexcept
43// Bit<Iter>& operator |=(bool b) const noexcept
44// Bit<Iter>& operator ^=(bool b) const noexcept
45// And, or and xor assignment operators.
46//
47//
48// Bit_pointer<T> is a pointer to a Bit<T> and supports all pointer operators. It is also an C++17 iterator
49// and therefore compatible with the STL. If a pointer operator causes a Bit_pointer to point beyond the
50// size of *T, it moves to another object as defined by the pointer arithmatic of container<T> and the
51// native byte order of *T (which is determined by the endedness of the hardware).
52//
53// Bit_pointer<T> assumes native byte-order. It has the following member functions:
54//
55// Bit_pointer<Iter>()
56// Default constructor of unspecified Bit_pointer<Iter> object.
57// Bit_pointer<Iter>(Iter offset)
58// Constructor of a Bit_pointer<Iter> object pointing to the first bit of *offset.
59// Bit_pointer<Iter>(Iter offset, std::ptrdiff_t n)
60// Constructor of a Bit_pointer<Iter> object pointing to the 'n'-th bit of *offset. If 'n' is
61// negative or larger than sizeof(*T), a bit in a preceeding / succeeding object is pointed to.
62// Bit_pointer<Iter>& operator =(Bit_pointer<Iter>) or =(Iter)
63// Assignment operators.
64// ++(); ()++; --(); ()--; += std::ptrdiff_t; -= std::ptrdiff_t; + std::size_t; - std::ptrdiff_t;
65// - Bit_pointer<Iter>; < ; <= ; > ; >= ; == ; !=
66// All the usual pointer arithmatic operators are allowed and return the expected values (i.e.
67// (references to) Bit_pointer<Iter> or std::ptrdiff_t
68//
69// Bit_range<T> defines a view of a range of Bit<T> in a container<*T>, providing an interface with the
70// STL. It also allows packing and unpacking integral types into and out of a Bit_range<T>. It has the
71// following member functions:
72//
73// Bit_range(Bit_pointer<Iter> const& bit_p, int const size)
74// Constructor; the number of bits is the range is fixed, but the range can be shifted to a new
75// position.
76// Bit_pointer<Iter> const begin()
77// Bit_pointer to the first bit of the range.
78// Bit_pointer<Iter> const end()
79// Bit pointer to the first bit beyond the range.
80// int const size()
81// Size of the Bit_range.
82// Bit_range<Iter>& next()
83// Alters the Bit_range by changing the start of the Bit_range to the bit pointed to by Bit_range::end().
84// The size of the Bit_range does not change.
85// template <typename T> operator T() const noexcept
86// Casts a Bit_range to an integral type. If T is a signed type, the value of the most significant bit
87// of the Bit_range determines the sign of the returned value. The Bit_range cannot have a size exceeding
88// that of the number of bits of T.
89// Bit_range& operator =(T const value)
90// Assigns an integral type to a Bit_range. The Bit_range cannot have a size exceeding that of the
91// number of bits of T.
92// Bit_range& operator |=(T const value)
93// Bitwise or-assigns an integral type to a Bit_range. The Bit_range cannot have a size exceeding that of the
94// number of bits of T. It is faster than direct assignment, provided all bits in the Bit_range are zero.
95// template <typename T_iter, typename T = typename std::iterator_traits<T_iter>::value_type> requires std::is_integral_v<T>
96// Bit_range& append_range(T_iter const from, T_iter const to) noexcept {
97// Assigns integral values defined by the range between the iterators 'from' and 'to' to the Bit_range and
98// succeeding Bit_ranges. Updates the Bit_range start to point to the next unassigned Bit_range. Before the
99// values defined by the iterator range 'from/to' are appended to the Bit_range, they are cast to the type T,
100// which defaults to the type of decltype(*from).
101// template <typename T_iter> requires std::is_integral_v<typename std::iterator_traits<T_iter>::value_type>
102// void get_range(T_iter const from, T_iter const to) noexcept
103// Extracts integral values from Bit_range and its succeeding Bit_ranges into the range defined by the
104// iterators from and to. Updates the Bit_range start to point to the next unassigned Bit_range.
105// If T_iter::value type has insufficient precision to store the extracted value, this value is clamped
106// to the maximum representable value (or minimum value in case T_iter::value is signed and underflow occurs).

121class Bit_pointer {
122 friend Bit<Iter>;
123 friend Bit_range<Iter>;
124
125 using Type = typename std::remove_reference<decltype(*Iter())>::type;
126
127public:

137 Bit_pointer() {
138 static_assert(std::is_integral_v<Type> && std::is_unsigned_v<Type>, "Bit_pointer requires a pointer to, or iterator of, an unsigned integral type");
139 };
140
146 Bit_pointer(Iter const offset) :
147 d_bit(0),
148 d_offset(offset) {
149 static_assert(std::is_integral_v<Type> && std::is_unsigned_v<Type>, "Bit_pointer requires a pointer to, or iterator of, an unsigned integral type");
150 };
151
159 Bit_pointer(Iter const offset, std::ptrdiff_t const bit) :
160 d_bit(int(bit & (sizeof(Type) * 8 - 1))),
161 d_offset(offset + (bit - d_bit) / int(sizeof(Type) * 8)) {
162 static_assert(std::is_integral_v<Type> && std::is_unsigned_v<Type>, "Bit_pointer requires a pointer to, or iterator of, an unsigned integral type");
163 };

244 d_bit += shift;

246 auto tmp = int(d_bit & std::ptrdiff_t(sizeof(Type) * 8 - 1));

248 d_bit = tmp;
249 }
250 return *this;

424public:
431 Bit(Iter location, std::ptrdiff_t const shift) : d_local_bitp(location, shift) {}
432
438 constexpr operator bool() const noexcept { return ((*d_bitp.d_offset) >> d_bitp.d_bit) & 1; }
439
445 constexpr Bit_pointer<Iter> operator&() const noexcept { return d_bitp; }
446
453 constexpr const Bit& operator=(bool const b) const noexcept { return *this ^= b ^ bool(*this); }
454
461 constexpr const Bit& operator=(Bit const& other) const noexcept { return *this = bool(other); }
462
469 constexpr const Bit& operator^=(bool const b) const noexcept { *d_bitp.d_offset ^= Type(b) << d_bitp.d_bit; return *this; }
470
477 constexpr const Bit& operator|=(bool const b) const noexcept { *d_bitp.d_offset |= Type(b) << d_bitp.d_bit; return *this; }
478
485 constexpr const Bit& operator&=(bool const b) const noexcept { *d_bitp.d_offset &= ~(Type(~b) << d_bitp.d_bit); return *this; }
486
490 constexpr void set() const noexcept { *d_bitp.d_offset |= Type(1) << d_bitp.d_bit; }
491
495 constexpr void reset() const noexcept { *d_bitp.d_offset &= ~(Type(1) << d_bitp.d_bit); }
496
500 constexpr void flip() const noexcept { *d_bitp.d_offset ^= Type(1) << d_bitp.d_bit; }
501
507 constexpr void swap(Bit& other) const noexcept {
508 Type tmp = (*d_bitp.d_offset ^ *other.d_bitp.d_offset) & ((Type(1)) << d_bitp.d_bit);
509 *d_bitp.d_offset ^= tmp;
510 *other.d_bitp.d_offset ^= tmp;
511 }

513private:

526template <typename Iter>
527class Bit_range {
528 using Type = typename std::remove_reference<decltype(*Iter())>::type;
529public:
530
538 Bit_range(Bit_pointer<Iter> const& bit_p, std::ptrdiff_t const size) :
539 d_bit_pointer(bit_p),
540 d_size(size) {};
541
548 Bit_range(Iter const start, std::ptrdiff_t const size) :
549 d_bit_pointer(start),
550 d_size(size) {};
551
557 Bit_pointer<Iter> const begin() const noexcept {return d_bit_pointer;}
558
564 Bit_pointer<Iter> const end() const noexcept {return d_bit_pointer + d_size;}
565
571 std::ptrdiff_t const size() const noexcept {return d_size;}
572
573
580 Bit_range<Iter>& next() noexcept {
581 d_bit_pointer.d_bit += size();
582 while (d_bit_pointer.d_bit >= (sizeof(Type) * 8)) {
583 d_bit_pointer.d_bit -= (sizeof(Type) * 8);
584 ++d_bit_pointer.d_offset;
585 }
586 return *this;
587 }
588
597 template <typename T>
598 operator T() const noexcept {
599 static_assert(std::is_integral<T>::value, "Bit_range must be cast to an integral type");
600 if (size() == 0)
601 return 0;
602 T result = T(*d_bit_pointer.d_offset >> d_bit_pointer.d_bit);
603 if constexpr (sizeof(T) <= sizeof(Type)) {
604 if ((d_bit_pointer.d_bit + size()) > sizeof(Type)*8)
605 result |= d_bit_pointer.d_offset[1] << (sizeof(Type)*8 - d_bit_pointer.d_bit);
606 }
607 else {
608 auto bp = d_bit_pointer.d_offset;
609 for (int i = sizeof(Type)*8 - d_bit_pointer.d_bit; i < size(); i += sizeof(Type)*8)
610 result |= *++bp << i;
611 }
612 T const mask = ((T(1) << d_size) - 1);
613 result &= mask;
614 if (std::is_signed_v<T> && (result & (T(1) << (size() - 1))))
615 return result | ~mask;
616 return result;
617 }
618
628 template <typename T>
629 Bit_range& operator |=(T value) noexcept {
630 static_assert(std::is_integral<T>::value, "Bit_range assignment requires an integral type");
631 if constexpr (std::is_signed_v<T>)
632 value &= ((T(1) << d_size) - 1);
633 if ((sizeof(T) <= sizeof(Type)) || (size() <= (sizeof(Type) * 8))) {
634 *d_bit_pointer.d_offset |= Type(value) << d_bit_pointer.d_bit;
635 int const shift_right = int(sizeof(Type)*8) - d_bit_pointer.d_bit;
636 if (shift_right < size())
637 d_bit_pointer.d_offset[1] |= value >> shift_right;
638 }
639 else {
640 auto bp = d_bit_pointer.d_offset;
641 *bp |= value << d_bit_pointer.d_bit;
642 value >>= int(sizeof(Type)*8) - d_bit_pointer.d_bit;
643 if constexpr (sizeof(Type) < sizeof(T))
644 do
645 *++bp |= value;
646 while (value >>= sizeof(Type)*8);
647 }
648 return *this;
649 }
650
659 template <typename T>
660 Bit_range& operator =(T value) noexcept {
661 static_assert(std::is_integral<T>::value, "Bit_range assignment requires an integral type");
662 value &= ((T(1) << d_size) - 1);
663 auto bp = d_bit_pointer.d_offset;
664 if constexpr (sizeof(T) <= sizeof(Type)) {
665 typename std::make_unsigned_t<typename Bit_pointer<Iter>::Type> mask = ((Type(1) << d_size) - 1);
666 *bp &= ~(mask << d_bit_pointer.d_bit);
667 *bp |= Type(value) << d_bit_pointer.d_bit;
668 int const shift_right = int(sizeof(Type)*8) - d_bit_pointer.d_bit;
669 if (shift_right < size()) {
670 *++bp &= ~(mask >> shift_right);
671 *bp |= value >> shift_right;
672 }
673 }
674 else {
675 typename std::make_unsigned_t<T> mask = ((T(1) << d_size) - 1);
676 *bp &= ~(mask << d_bit_pointer.d_bit);
677 *bp |= value << d_bit_pointer.d_bit;
678 int const shift_right = int(sizeof(Type)*8) - d_bit_pointer.d_bit;
679 value >>= shift_right;
680 int bits_left = int(size()) - shift_right;
681 for ( ; bits_left > int(sizeof(Type)*8); bits_left -= int(sizeof(Type)*8), value >>= int(sizeof(Type)*8))
682 *++bp = value;
683 if (bits_left) {
684 *++bp &= ~(mask >> (size() - bits_left));
685 *bp |= value;
686 }
687 }
688 return *this;
689 }

700 template <typename T_iter> requires std::is_integral_v<typename std::iterator_traits<T_iter>::value_type>
701 Bit_range& append_range(T_iter const from, T_iter const to) noexcept {
702 using T = typename std::iterator_traits<T_iter>::value_type;
703 if (this->size()) {
704 Type buffer = *d_bit_pointer.d_offset;
705 std::conditional_t<(sizeof(Type) > sizeof(T)), Type, T> value;
706 for (auto p = from; p != to; ++p) {
707 if constexpr (std::is_signed_v<T>)
708 value = static_cast<T>(*p) & ((T(1) << d_size) - 1);
709 else
710 value = static_cast<T>(*p);
711 buffer |= value << d_bit_pointer.d_bit;
712 d_bit_pointer.d_bit += this->size();
713 if (d_bit_pointer.d_bit >= (sizeof(Type) * 8)) {
714 *(d_bit_pointer.d_offset++) = buffer;
715 d_bit_pointer.d_bit -= sizeof(Type) * 8;
716 buffer = value >>= (this->size() - d_bit_pointer.d_bit);
717 if constexpr (sizeof(T) > sizeof(Type)) {
718 while (d_bit_pointer.d_bit >= (sizeof(Type) * 8)) {
719 *(d_bit_pointer.d_offset++) = buffer;
720 d_bit_pointer.d_bit -= sizeof(Type) * 8;
721 buffer = value >>= sizeof(Type) * 8;
722 }
723 }
724
725 }
726 }
727 *d_bit_pointer.d_offset = buffer;
728 }
729 return *this;
730 }

742 template <typename T_iter> requires std::is_integral_v<typename std::iterator_traits<T_iter>::value_type>
743 void get_range(T_iter const from, T_iter const to) noexcept {
744 using T = typename std::iterator_traits<T_iter>::value_type;
745 if (size() == 0)
746 std::fill(from, to, 0);
747 else if (sizeof(T) * 8 < this->size()) {
748 if constexpr (std::is_unsigned_v<T>) {
749 std::uint64_t arr[to - from];
750 get_range(arr, arr + (to - from));
751 constexpr std::uint64_t upper = static_cast<std::int64_t>(std::numeric_limits<T>::max());
752 for (int i=0; i != (to - from); ++i)
753 from[i] = static_cast<T>(std::min(arr[i], upper));
754 }
755 else {
756 std::int64_t arr[to - from];
757 get_range(arr, arr + (to - from));
758 constexpr std::int64_t lower = static_cast<std::int64_t>(std::numeric_limits<T>::min());
759 constexpr std::int64_t upper = static_cast<std::int64_t>(std::numeric_limits<T>::max());
760 for (int i=0; i != (to - from); ++i)
761 from[i] = static_cast<T>(std::clamp(arr[i], lower, upper));
762 }
763 }
764 else {
765 T const mask = ((T(1) << d_size) - 1);
766 std::remove_cv_t<Type> buffer = *d_bit_pointer.d_offset >> d_bit_pointer.d_bit;
767 for (auto p = from; p != to; ++p) {
768 T result = T(buffer);
769 buffer >>= this->size();
770 d_bit_pointer.d_bit += this->size();
771 if constexpr (sizeof(T) > sizeof(Type))
772 while (d_bit_pointer.d_bit >= (sizeof(Type) * 8)) {
773 buffer = *++d_bit_pointer.d_offset;
774 d_bit_pointer.d_bit -= sizeof(Type) * 8;
775 result |= buffer << (this->size() - d_bit_pointer.d_bit);
776 buffer >>= d_bit_pointer.d_bit;
777 }
778 else if (d_bit_pointer.d_bit >= sizeof(Type) * 8) {
779 buffer = *++d_bit_pointer.d_offset;
780 d_bit_pointer.d_bit -= sizeof(Type) * 8;
781 result |= buffer << (this->size() - d_bit_pointer.d_bit);
782 buffer >>= d_bit_pointer.d_bit;
783 }
784 if constexpr (std::is_unsigned_v<T>)
785 *p = result & mask;
786 else if (result & (T(1) << (size() - 1)))
787 *p = (result & mask) | ~mask;
788 else
789 *p = result & mask;
790 }
791 }
792 }
